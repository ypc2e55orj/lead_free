59 static void LINE_UpdateMarkerState()
60 {
61  bool isStartGoal = ADC_GetLineSensorValue(MARKER_SENSOR_RIGHT) > lineMarkerCalibrateValue.threshold[MARKER_SENSOR_RIGHT];
62  bool isCurvature = ADC_GetLineSensorValue(MARKER_SENSOR_LEFT) > lineMarkerCalibrateValue.threshold[MARKER_SENSOR_LEFT];
63 
64  if (isStartGoal && isCurvature)
65  {
66  if (lineMarkerStartGoalState != STARTGOAL_MARKER_GOAL_PASSED)
67  {
68  lineMarkerStartGoalState = STARTGOAL_MARKER_IGNORING;
69  }
70  if (lineMarkerCurvatureState != CURVATURE_MARKER_PASSED)
71  {
72  lineMarkerCurvatureState = CURVATURE_MARKER_IGNORING;
73  }
74  }
75  switch (lineMarkerStartGoalState)
76  {
77  default:
78  case STARTGOAL_MARKER_START_WAITING:
79  if (isStartGoal)
80  {
81  lineMarkerStartGoalState = STARTGOAL_MARKER_START_PASSING;
82  }
83  break;
84  case STARTGOAL_MARKER_START_PASSING:
85  if (!isStartGoal)
86  {
87  lineMarkerStartGoalState = STARTGOAL_MARKER_START_PASSED;
88  }
89  break;
90  case STARTGOAL_MARKER_START_PASSED:
91  lineMarkerStartGoalState = STARTGOAL_MARKER_GOAL_WAITING;
92  break;
93  case STARTGOAL_MARKER_GOAL_WAITING:
94  if (isStartGoal)
95  {
96  lineMarkerStartGoalState = STARTGOAL_MARKER_GOAL_PASSING;
97  }
98  break;
99  case STARTGOAL_MARKER_GOAL_PASSING:
100  if (!isStartGoal)
101  {
102  lineMarkerStartGoalState = STARTGOAL_MARKER_GOAL_PASSED;
103  }
104  break;
105  case STARTGOAL_MARKER_GOAL_PASSED:
106  break;
107  case STARTGOAL_MARKER_IGNORING:
108  if (!isStartGoal)
109  {
110  lineMarkerStartGoalState = STARTGOAL_MARKER_GOAL_WAITING;
111  }
112  break;
113  }
114  switch (lineMarkerCurvatureState)
115  {
116  default:
117  case CURVATURE_MARKER_WAITING:
118  if (isCurvature)
119  {
120  lineMarkerCurvatureState = CURVATURE_MARKER_PASSING;
121  }
122  break;
123  case CURVATURE_MARKER_PASSING:
124  if (!isCurvature)
125  {
126  lineMarkerCurvatureState = CURVATURE_MARKER_PASSED;
127  }
128  break;
129  case CURVATURE_MARKER_PASSED:
130  lineMarkerCurvatureState = CURVATURE_MARKER_WAITING;
131  break;
132  case CURVATURE_MARKER_IGNORING:
133  if (!isCurvature)
134  {
135  lineMarkerCurvatureState = CURVATURE_MARKER_WAITING;
136  }
137  break;
138  }

143 static void LINE_UpdateAngularVelocity()
144 {
145  if (!lineEnableFeedback)
146  {
147  lineAngularVelocity = 0.0f;
148  return;
149  }
150 
151  bool isRightOut = lineMarkerValue[LINE_SENSOR_RIGHT_OUT] > lineMarkerCalibrateValue.threshold[LINE_SENSOR_RIGHT_OUT];
152  bool isRighIn = lineMarkerValue[LINE_SENSOR_RIGHT_IN] > lineMarkerCalibrateValue.threshold[LINE_SENSOR_RIGHT_IN];
153  bool isLeftIn = lineMarkerValue[LINE_SENSOR_LEFT_IN] > lineMarkerCalibrateValue.threshold[LINE_SENSOR_LEFT_IN];
154  bool isLeftOut = lineMarkerValue[LINE_SENSOR_LEFT_OUT] > lineMarkerCalibrateValue.threshold[LINE_SENSOR_LEFT_OUT];
155 
156  if ((isRightOut && isLeftIn) || (isRighIn && isLeftOut))
157  {
158  // Cross line
159  lineAngularVelocity = 0.0f;
160  lineState = LINE_STATE_CROSS;
161  }
162  else if (!isRightOut && !isRighIn && !isLeftIn && !isLeftOut)
163  {
164  // Course out
165  lineAngularVelocity = 0.0f;
166  lineState = LINE_STATE_COURSE_OUT;
167  }
168  else
169  {
170  // Good
171  lineOutError = (lineMarkerValue[LINE_SENSOR_RIGHT_OUT] + lineMarkerCalibrateValue.lineOffsetOut) - lineMarkerValue[LINE_SENSOR_LEFT_OUT];
172  lineInError = (lineMarkerValue[LINE_SENSOR_RIGHT_IN] + lineMarkerCalibrateValue.lineOffsetIn) - lineMarkerValue[LINE_SENSOR_LEFT_IN];
173  lineAngularVelocity = PID_Update(&lineAngularVelocityPid, 0.0f, (lineInError + 2.0f * lineOutError), 1.0f);
174  lineState = LINE_STATE_GOOD;
175  }
