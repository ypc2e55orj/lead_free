60 static void LINE_UpdateMarkerState()
61 {
62  bool isStartGoal = ADC_GetLineSensorValue(MARKER_SENSOR_RIGHT) > lineMarkerCalibrateValue.threshold[MARKER_SENSOR_RIGHT];
63  bool isCurvature = ADC_GetLineSensorValue(MARKER_SENSOR_LEFT) > lineMarkerCalibrateValue.threshold[MARKER_SENSOR_LEFT];
64 
65  if (isStartGoal && isCurvature)
66  {
67  if (lineMarkerStartGoalState != STARTGOAL_MARKER_GOAL_PASSED)
68  {
69  lineMarkerStartGoalState = STARTGOAL_MARKER_IGNORING;
70  }
71  if (lineMarkerCurvatureState != CURVATURE_MARKER_PASSED)
72  {
73  lineMarkerCurvatureState = CURVATURE_MARKER_IGNORING;
74  }
75  }
76  switch (lineMarkerStartGoalState)
77  {
78  default:
79  case STARTGOAL_MARKER_START_WAITING:
80  if (isStartGoal)
81  {
82  lineMarkerStartGoalState = STARTGOAL_MARKER_START_PASSING;
83  }
84  break;
85  case STARTGOAL_MARKER_START_PASSING:
86  if (!isStartGoal)
87  {
88  lineMarkerStartGoalState = STARTGOAL_MARKER_START_PASSED;
89  }
90  break;
91  case STARTGOAL_MARKER_START_PASSED:
92  lineMarkerStartGoalState = STARTGOAL_MARKER_GOAL_WAITING;
93  break;
94  case STARTGOAL_MARKER_GOAL_WAITING:
95  if (isStartGoal)
96  {
97  lineMarkerStartGoalState = STARTGOAL_MARKER_GOAL_PASSING;
98  }
99  break;
100  case STARTGOAL_MARKER_GOAL_PASSING:
101  if (!isStartGoal)
102  {
103  lineMarkerStartGoalState = STARTGOAL_MARKER_GOAL_PASSED;
104  }
105  break;
106  case STARTGOAL_MARKER_GOAL_PASSED:
107  break;
108  case STARTGOAL_MARKER_IGNORING:
109  if (!isStartGoal)
110  {
111  lineMarkerStartGoalState = STARTGOAL_MARKER_GOAL_WAITING;
112  }
113  break;
114  }
115  switch (lineMarkerCurvatureState)
116  {
117  default:
118  case CURVATURE_MARKER_WAITING:
119  if (isCurvature)
120  {
121  lineMarkerCurvatureState = CURVATURE_MARKER_PASSING;
122  }
123  break;
124  case CURVATURE_MARKER_PASSING:
125  if (!isCurvature)
126  {
127  lineMarkerCurvatureState = CURVATURE_MARKER_PASSED;
128  }
129  break;
130  case CURVATURE_MARKER_PASSED:
131  lineMarkerCurvatureState = CURVATURE_MARKER_WAITING;
132  break;
133  case CURVATURE_MARKER_IGNORING:
134  if (!isCurvature)
135  {
136  lineMarkerCurvatureState = CURVATURE_MARKER_WAITING;
137  }
138  break;
139  }

144 static void LINE_UpdateAngularVelocity()
145 {
146  if (!lineEnableFeedback)
147  {
148  lineAngularVelocity = 0.0f;
149  return;
150  }
151 
152  bool isRightOut = lineMarkerValue[LINE_SENSOR_RIGHT_OUT] > lineMarkerCalibrateValue.threshold[LINE_SENSOR_RIGHT_OUT];
153  bool isRighIn = lineMarkerValue[LINE_SENSOR_RIGHT_IN] > lineMarkerCalibrateValue.threshold[LINE_SENSOR_RIGHT_IN];
154  bool isLeftIn = lineMarkerValue[LINE_SENSOR_LEFT_IN] > lineMarkerCalibrateValue.threshold[LINE_SENSOR_LEFT_IN];
155  bool isLeftOut = lineMarkerValue[LINE_SENSOR_LEFT_OUT] > lineMarkerCalibrateValue.threshold[LINE_SENSOR_LEFT_OUT];
156 
157  if (!isRightOut && !isRighIn && !isLeftIn && !isLeftOut)
158  {
159  // Course out
160  lineAngularVelocity = 0.0f;
161  lineState = LINE_STATE_COURSE_OUT;
162  }
163  else
164  {
165  // Good
166  lineOutError = (lineMarkerValue[LINE_SENSOR_RIGHT_OUT] + lineMarkerCalibrateValue.lineOffsetOut) - lineMarkerValue[LINE_SENSOR_LEFT_OUT];
167  lineInError = (lineMarkerValue[LINE_SENSOR_RIGHT_IN] + lineMarkerCalibrateValue.lineOffsetIn) - lineMarkerValue[LINE_SENSOR_LEFT_IN];
168  lineAngularVelocity = PID_Update(&lineAngularVelocityPid, 0.0f, (lineInError + 2.0f * lineOutError), 1.0f);
169  lineState = LINE_STATE_GOOD;
170  }
